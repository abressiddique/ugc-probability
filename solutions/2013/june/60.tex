For $x^2+2U x+V=0$ to have real roots,
\begin{align}
    b^2-4ac\geq0\\
    \brak{2U}^2-4\brak{1}\brak{V}\geq0\\
    U^2\geq V
\end{align}
\begin{align}
    \pr{U^2\geq V}=1-\pr{U^2<V}
\end{align}
